# Modo Duelo PvP — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement real-time 2-player PvP duel mode where both players hunt words on a shared synchronized board and launch powers with cooldowns via Socket.io.

**Architecture:** Node.js + Socket.io server in `server/` subfolder, deployed as Docker container on existing VPS (port 3001). Client connects via `socket.io-client`. Board generated by Jugador A (client), sent to server as authoritative word list. Server handles race conditions, power validation, and cooldowns. Four new client screens: DuelMenu, DuelLobby, DuelGame, DuelResult.

**Tech Stack:** Node.js 20, Socket.io 4, TypeScript, ts-node (dev), Docker (prod), Nginx reverse proxy, socket.io-client (Vite client)

**Spec:** `docs/superpowers/specs/2026-06-13-duelo-pvp-design.md`

```typescript
import type { RoomState, PlayerState, PlayerSlot, DuelWord, PowerId } from './types'

const POWER_UNLOCK_EVERY = 3   // every N catches, earn a power
const MAX_INVENTORY = 2

const ALL_POWERS: PowerId[] = ['FREEZE', 'STEAL', 'DECOY', 'SHIELD', 'DOUBLE']

function randomPower(): PowerId {
  return ALL_POWERS[Math.floor(Math.random() * ALL_POWERS.length)]
}

export function createRoom(code: string): RoomState {
  return {
    code,
    players: new Map(),
    words: [],
    duracion: 180,
    cat: '',
    nivel: 1,
    startedAt: null,
    timerHandle: null,
    phase: 'waiting',
  }
}

export function addPlayer(room: RoomState, socketId: string, nombre: string): PlayerSlot | null {
  if (room.players.size >= 2) return null
  const slot: PlayerSlot = room.players.has('A') ? 'B' : 'A'
  room.players.set(slot, {
    socketId,
    nombre,
    slot,
    score: 0,
    powerInventory: [],
    activeEffects: [],
    catchCount: 0,
    ready: false,
  })
  return slot
}

export function removePlayer(room: RoomState, socketId: string): PlayerSlot | null {
  for (const [slot, p] of room.players) {
    if (p.socketId === socketId) {
      room.players.delete(slot)
      return slot
    }
  }
  return null
}

export function getPlayerBySocket(room: RoomState, socketId: string): PlayerState | null {
  for (const p of room.players.values()) {
    if (p.socketId === socketId) return p
  }
  return null
}

export function getRival(room: RoomState, slot: PlayerSlot): PlayerState | null {
  const rivalSlot: PlayerSlot = slot === 'A' ? 'B' : 'A'
  return room.players.get(rivalSlot) ?? null
}

export function setBoard(room: RoomState, words: DuelWord[], duracion: number, cat: string, nivel: number): void {
  room.words = words
  room.duracion = duracion
  room.cat = cat
  room.nivel = nivel
}

export function markReady(room: RoomState, socketId: string): boolean {
  const p = getPlayerBySocket(room, socketId)
  if (!p) return false
  p.ready = true
  return true
}

export function allReady(room: RoomState): boolean {
  if (room.players.size < 2) return false
  return [...room.players.values()].every(p => p.ready)
}

export interface CatchResult {
  success: boolean
  alreadyTaken: boolean
  pointsEarned: number
  stolenByRival: boolean  // rival STEAL effect was active
  powerEarned: PowerId | null
}

export function catchWord(room: RoomState, socketId: string, wordId: string): CatchResult {
  const catcher = getPlayerBySocket(room, socketId)
  if (!catcher) return { success: false, alreadyTaken: false, pointsEarned: 0, stolenByRival: false, powerEarned: null }

  const word = room.words.find(w => w.id === wordId)
  if (!word) return { success: false, alreadyTaken: false, pointsEarned: 0, stolenByRival: false, powerEarned: null }
  if (word.takenBy !== null) return { success: false, alreadyTaken: true, pointsEarned: 0, stolenByRival: false, powerEarned: null }

  // Check FREEZE effect on catcher
  const frozen = catcher.activeEffects.some(e => e.type === 'FREEZE' && (e.expiresAt === null || e.expiresAt > Date.now()))
  if (frozen) return { success: false, alreadyTaken: false, pointsEarned: 0, stolenByRival: false, powerEarned: null }

  word.takenBy = catcher.slot

  const basePoints = word.isCorrect ? (10 * room.nivel) : 0
  const hasDouble = consumeEventEffect(catcher, 'DOUBLE')
  const points = basePoints * (hasDouble ? 2 : 1)

  // Check rival STEAL effect
  const rival = getRival(room, catcher.slot)
  let stolenByRival = false
  if (rival) {
    const rivalHasSteal = consumeEventEffect(rival, 'STEAL')
    if (rivalHasSteal) {
      rival.score += points
      stolenByRival = true
    } else {
      catcher.score += points
    }
  } else {
    catcher.score += points
  }

  catcher.catchCount++
  let powerEarned: PowerId | null = null
  if (catcher.catchCount % POWER_UNLOCK_EVERY === 0 && catcher.powerInventory.length < MAX_INVENTORY) {
    powerEarned = randomPower()
    catcher.powerInventory.push(powerEarned)
  }

  return { success: true, alreadyTaken: false, pointsEarned: points, stolenByRival, powerEarned }
}

function consumeEventEffect(player: PlayerState, type: PowerId): boolean {
  const idx = player.activeEffects.findIndex(e => e.type === type && e.capturesRemaining !== null && e.capturesRemaining > 0)
  if (idx === -1) return false
  player.activeEffects[idx].capturesRemaining!--
  if (player.activeEffects[idx].capturesRemaining === 0) {
    player.activeEffects.splice(idx, 1)
  }
  return true
}

export function pruneExpiredEffects(room: RoomState): void {
  const now = Date.now()
  for (const p of room.players.values()) {
    p.activeEffects = p.activeEffects.filter(e => {
      if (e.expiresAt !== null && e.expiresAt <= now) return false
      return true
    })
  }
}

export function getScores(room: RoomState): { A: number; B: number } {
  const A = room.players.get('A')?.score ?? 0
  const B = room.players.get('B')?.score ?? 0
  return { A, B }
}

export function determineWinner(room: RoomState): 'A' | 'B' | 'draw' {
  const { A, B } = getScores(room)
  if (A > B) return 'A'
  if (B > A) return 'B'
  return 'draw'
}
```

```typescript
import type { PowerId, PlayerState, PlayerSlot, RoomState, ActiveEffect } from './types'
import { getPlayerBySocket, getRival, pruneExpiredEffects } from './DuelRoom'

interface CooldownEntry {
  lastUsed: number
}

// cooldown in ms per power
const COOLDOWNS: Record<PowerId, number> = {
  FREEZE: 25000,
  STEAL:  20000,
  DECOY:  30000,
  SHIELD: 22000,
  DOUBLE: 18000,
}

// effect duration in ms (null = event-based / consumed on next capture)
const EFFECT_DURATION: Record<PowerId, number | null> = {
  FREEZE: 3000,
  STEAL:  null,   // consumed on rival's next capture
  DECOY:  8000,
  SHIELD: null,   // consumed when blocking a power
  DOUBLE: null,   // consumed on catcher's next capture
}

// map: socketId -> Map<PowerId, CooldownEntry>
const cooldowns = new Map<string, Map<PowerId, CooldownEntry>>()

export function registerSocket(socketId: string): void {
  cooldowns.set(socketId, new Map())
}

export function unregisterSocket(socketId: string): void {
  cooldowns.delete(socketId)
}

export type PowerResult =
  | { ok: true; blocked: boolean; decoyWords?: string[] }
  | { ok: false; reason: 'cooldown' | 'no_power' | 'no_rival' | 'frozen' }

export function usePower(room: RoomState, socketId: string, powerId: PowerId): PowerResult {
  const caster = getPlayerBySocket(room, socketId)
  if (!caster) return { ok: false, reason: 'no_rival' }

  // Check inventory
  const invIdx = caster.powerInventory.indexOf(powerId)
  if (invIdx === -1) return { ok: false, reason: 'no_power' }

  // Check cooldown
  const socketCooldowns = cooldowns.get(socketId) ?? new Map()
  const entry = socketCooldowns.get(powerId)
  const now = Date.now()
  if (entry && now - entry.lastUsed < COOLDOWNS[powerId]) {
    return { ok: false, reason: 'cooldown' }
  }

  const rival = getRival(room, caster.slot)
  if (!rival && powerId !== 'DOUBLE' && powerId !== 'SHIELD') {
    return { ok: false, reason: 'no_rival' }
  }

  // DOUBLE/SHIELD affect caster, not rival — apply directly
  if (powerId === 'DOUBLE') {
    consumeFromInventory(caster, powerId)
    recordCooldown(socketId, powerId)
    applyEffect(caster, { type: 'DOUBLE', expiresAt: null, capturesRemaining: 1 })
    return { ok: true, blocked: false }
  }

  if (powerId === 'SHIELD') {
    consumeFromInventory(caster, powerId)
    recordCooldown(socketId, powerId)
    applyEffect(caster, { type: 'SHIELD', expiresAt: null, capturesRemaining: 1 })
    return { ok: true, blocked: false }
  }

  // Check if rival has SHIELD active
  const rivalShieldIdx = rival!.activeEffects.findIndex(e => e.type === 'SHIELD' && e.capturesRemaining !== null && e.capturesRemaining > 0)
  if (rivalShieldIdx !== -1) {
    rival!.activeEffects.splice(rivalShieldIdx, 1)
    consumeFromInventory(caster, powerId)
    recordCooldown(socketId, powerId)
    return { ok: true, blocked: true }
  }

  // Apply effect to rival
  consumeFromInventory(caster, powerId)
  recordCooldown(socketId, powerId)

  const duration = EFFECT_DURATION[powerId]
  const effect: ActiveEffect = {
    type: powerId,
    expiresAt: duration !== null ? now + duration : null,
    capturesRemaining: powerId === 'STEAL' ? 1 : null,
  }
  applyEffect(rival!, effect)

  if (powerId === 'DECOY') {
    const decoyWords = generateDecoyWords(room.words.filter(w => w.takenBy === null).map(w => w.text))
    return { ok: true, blocked: false, decoyWords }
  }

  return { ok: true, blocked: false }
}

function consumeFromInventory(player: PlayerState, powerId: PowerId): void {
  const idx = player.powerInventory.indexOf(powerId)
  if (idx !== -1) player.powerInventory.splice(idx, 1)
}

function recordCooldown(socketId: string, powerId: PowerId): void {
  const map = cooldowns.get(socketId) ?? new Map()
  map.set(powerId, { lastUsed: Date.now() })
  cooldowns.set(socketId, map)
}

function applyEffect(player: PlayerState, effect: ActiveEffect): void {
  // Remove existing same-type effect before applying new one
  player.activeEffects = player.activeEffects.filter(e => e.type !== effect.type)
  player.activeEffects.push(effect)
}

function generateDecoyWords(realWords: string[]): string[] {
  // Simple decoys: take real words and scramble them, or use fixed list
  const decoys = ['correr','grande','rápido','hermoso','cielo','luna','agua','fuego']
  const shuffled = decoys.sort(() => Math.random() - 0.5)
  return shuffled.slice(0, 4)
}

export function getCooldownRemaining(socketId: string, powerId: PowerId): number {
  const map = cooldowns.get(socketId)
  if (!map) return 0
  const entry = map.get(powerId)
  if (!entry) return 0
  const remaining = COOLDOWNS[powerId] - (Date.now() - entry.lastUsed)
  return Math.max(0, remaining)
}
```

```typescript
import { createServer } from 'http'
import { Server, Socket } from 'socket.io'
import type { RoomState, DuelWord } from './types'
import {
  createRoom, addPlayer, removePlayer, getPlayerBySocket,
  getRival, setBoard, markReady, allReady, catchWord,
  getScores, determineWinner, pruneExpiredEffects
} from './DuelRoom'
import { registerSocket, unregisterSocket, usePower } from './PowerManager'

const PORT = parseInt(process.env.PORT ?? '3001', 10)
const ROOM_IDLE_TIMEOUT = 5 * 60 * 1000  // 5 min

const httpServer = createServer()
const io = new Server(httpServer, {
  cors: { origin: '*', methods: ['GET', 'POST'] },
})

// roomCode -> RoomState
const rooms = new Map<string, RoomState>()
// socketId -> roomCode
const socketRoom = new Map<string, string>()

function generateCode(): string {
  const chars = 'ABCDEFGHJKLMNPQRSTUVWXYZ23456789'
  let code = ''
  for (let i = 0; i < 4; i++) code += chars[Math.floor(Math.random() * chars.length)]
  return code
}

function getUniqueCode(): string {
  let code = generateCode()
  while (rooms.has(code)) code = generateCode()
  return code
}

function endDuel(room: RoomState, reason: 'time' | 'all_words' | 'disconnect'): void {
  if (room.phase === 'ended') return
  room.phase = 'ended'
  if (room.timerHandle) clearTimeout(room.timerHandle)

  const winner = determineWinner(room)
  const scores = getScores(room)

  io.to(room.code).emit('duel_end', { winner, scores, reason })

  // Cleanup room after 30s
  setTimeout(() => rooms.delete(room.code), 30000)
}

io.on('connection', (socket: Socket) => {
  registerSocket(socket.id)

  // ── Create duel ──────────────────────────────────────────────────────────
  socket.on('create_duel', (data: {
    nombre: string
    cat: string
    nivel: number
    duracion: number
    words: Array<{ id: string; text: string; isCorrect: boolean }>
  }) => {
    const code = getUniqueCode()
    const room = createRoom(code)

    const duelWords: DuelWord[] = data.words.map(w => ({ ...w, takenBy: null }))
    setBoard(room, duelWords, data.duracion, data.cat, data.nivel)

    const slot = addPlayer(room, socket.id, data.nombre)
    if (!slot) { socket.emit('error', 'Could not create room'); return }

    rooms.set(code, room)
    socketRoom.set(socket.id, code)
    socket.join(code)
    socket.emit('duel_created', { code, slot })

    // Auto-destroy idle room
    room.timerHandle = setTimeout(() => {
      if (room.phase === 'waiting') rooms.delete(code)
    }, ROOM_IDLE_TIMEOUT)
  })

  // ── Join duel ─────────────────────────────────────────────────────────────
  socket.on('join_duel', (data: { code: string; nombre: string }) => {
    const room = rooms.get(data.code.toUpperCase())
    if (!room) { socket.emit('error', 'Sala no encontrada'); return }
    if (room.phase !== 'waiting') { socket.emit('error', 'La partida ya comenzó'); return }
    if (room.players.size >= 2) { socket.emit('error', 'Sala llena'); return }

    const slot = addPlayer(room, socket.id, data.nombre)
    if (!slot) { socket.emit('error', 'No se pudo unir'); return }

    socketRoom.set(socket.id, data.code.toUpperCase())
    socket.join(data.code.toUpperCase())

    const rival = getRival(room, slot)!
    socket.emit('duel_joined', { slot, rivalNombre: rival.nombre })
    socket.to(data.code.toUpperCase()).emit('rival_joined', { nombre: data.nombre })
  })

  // ── Player ready ──────────────────────────────────────────────────────────
  socket.on('player_ready', () => {
    const code = socketRoom.get(socket.id)
    if (!code) return
    const room = rooms.get(code)
    if (!room || room.phase !== 'waiting') return

    markReady(room, socket.id)

    if (allReady(room)) {
      if (room.timerHandle) clearTimeout(room.timerHandle)
      room.phase = 'countdown'

      io.to(code).emit('countdown_start')

      setTimeout(() => {
        room.phase = 'playing'
        room.startedAt = Date.now()
        io.to(code).emit('duel_start', {
          words: room.words,
          duracion: room.duracion,
          cat: room.cat,
          nivel: room.nivel,
        })

        room.timerHandle = setTimeout(() => endDuel(room, 'time'), room.duracion * 1000)
      }, 3000)
    }
  })

  // ── Word caught ───────────────────────────────────────────────────────────
  socket.on('word_caught', (wordId: string) => {
    const code = socketRoom.get(socket.id)
    if (!code) return
    const room = rooms.get(code)
    if (!room || room.phase !== 'playing') return

    pruneExpiredEffects(room)
    const result = catchWord(room, socket.id, wordId)

    if (result.alreadyTaken) {
      socket.emit('word_already_taken', wordId)
      return
    }

    if (!result.success) {
      // frozen or invalid
      return
    }

    const catcher = getPlayerBySocket(room, socket.id)!
    io.to(code).emit('word_taken', {
      wordId,
      bySlot: catcher.slot,
      stolenByRival: result.stolenByRival,
      scores: getScores(room),
    })

    if (result.powerEarned) {
      socket.emit('power_earned', result.powerEarned)
    }

    // Check all words taken
    if (room.words.every(w => w.takenBy !== null)) {
      endDuel(room, 'all_words')
    }
  })

  // ── Use power ─────────────────────────────────────────────────────────────
  socket.on('use_power', (powerId: string) => {
    const code = socketRoom.get(socket.id)
    if (!code) return
    const room = rooms.get(code)
    if (!room || room.phase !== 'playing') return

    const validPowers = ['FREEZE', 'STEAL', 'DECOY', 'SHIELD', 'DOUBLE']
    if (!validPowers.includes(powerId)) return

    const result = usePower(room, socket.id, powerId as any)

    if (!result.ok) {
      socket.emit('power_failed', { powerId, reason: result.reason })
      return
    }

    const caster = getPlayerBySocket(room, socket.id)!
    const rival = getRival(room, caster.slot)

    if (result.blocked) {
      socket.emit('power_blocked', powerId)
      if (rival) {
        io.to(rival.socketId).emit('shield_consumed', powerId)
      }
    } else {
      // Notify both
      socket.emit('power_used', powerId)

      if (rival && powerId !== 'DOUBLE' && powerId !== 'SHIELD') {
        io.to(rival.socketId).emit('power_effect', {
          powerId,
          fromSlot: caster.slot,
          decoyWords: (result as any).decoyWords ?? [],
        })
      }
    }
  })

  // ── Rematch ───────────────────────────────────────────────────────────────
  socket.on('request_rematch', () => {
    const code = socketRoom.get(socket.id)
    if (!code) return
    socket.to(code).emit('rematch_requested')
  })

  socket.on('accept_rematch', () => {
    const code = socketRoom.get(socket.id)
    if (!code) return
    const room = rooms.get(code)
    if (!room || room.phase !== 'ended') return

    // Reset room for rematch
    room.phase = 'waiting'
    room.startedAt = null
    room.words.forEach(w => w.takenBy = null)
    for (const p of room.players.values()) {
      p.score = 0
      p.powerInventory = []
      p.activeEffects = []
      p.catchCount = 0
      p.ready = false
    }
    io.to(code).emit('rematch_start')
  })

  // ── Disconnect ────────────────────────────────────────────────────────────
  socket.on('disconnect', () => {
    unregisterSocket(socket.id)
    const code = socketRoom.get(socket.id)
    if (!code) return
    socketRoom.delete(socket.id)

    const room = rooms.get(code)
    if (!room) return

    removePlayer(room, socket.id)

    if (room.phase === 'playing') {
      endDuel(room, 'disconnect')
      socket.to(code).emit('rival_disconnected')
    } else if (room.players.size === 0) {
      rooms.delete(code)
    }
  })
})

httpServer.listen(PORT, () => {
```

```typescript
import { io, Socket } from 'socket.io-client'
import type { DuelWord, DuelScores, PowerId, PlayerSlot } from '../types/duel'

const SERVER_URL = 'https://aulaplay.duckdns.org'
const SOCKET_PATH = '/duel/socket.io'

export type DuelEventMap = {
  duel_created:      (data: { code: string; slot: PlayerSlot }) => void
  duel_joined:       (data: { slot: PlayerSlot; rivalNombre: string }) => void
  rival_joined:      (data: { nombre: string }) => void
  countdown_start:   () => void
  duel_start:        (data: { words: DuelWord[]; duracion: number; cat: string; nivel: number }) => void
  word_taken:        (data: { wordId: string; bySlot: PlayerSlot; stolenByRival: boolean; scores: DuelScores }) => void
  word_already_taken:(wordId: string) => void
  power_earned:      (powerId: PowerId) => void
  power_used:        (powerId: PowerId) => void
  power_effect:      (data: { powerId: PowerId; fromSlot: PlayerSlot; decoyWords: string[] }) => void
  power_blocked:     (powerId: PowerId) => void
  power_failed:      (data: { powerId: PowerId; reason: string }) => void
  shield_consumed:   (powerId: PowerId) => void
  duel_end:          (data: { winner: 'A' | 'B' | 'draw'; scores: DuelScores; reason: string }) => void
  rival_disconnected:() => void
  rematch_requested: () => void
  rematch_start:     () => void
  error:             (msg: string) => void
  connect:           () => void
  disconnect:        () => void
}

class DuelService {
  private socket: Socket | null = null
  private handlers = new Map<string, Set<Function>>()

  connect(): void {
    if (this.socket?.connected) return
    this.socket = io(SERVER_URL, {
      path: SOCKET_PATH,
      transports: ['websocket'],
    })
    this.socket.onAny((event: string, ...args: any[]) => {
      this.handlers.get(event)?.forEach(h => h(...args))
    })
  }

  disconnect(): void {
    this.socket?.disconnect()
    this.socket = null
  }

  on<K extends keyof DuelEventMap>(event: K, handler: DuelEventMap[K]): void {
    if (!this.handlers.has(event)) this.handlers.set(event, new Set())
    this.handlers.get(event)!.add(handler as Function)
  }

  off<K extends keyof DuelEventMap>(event: K, handler: DuelEventMap[K]): void {
    this.handlers.get(event)?.delete(handler as Function)
  }

  createDuel(data: { nombre: string; cat: string; nivel: number; duracion: number; words: Array<{ id: string; text: string; isCorrect: boolean }> }): void {
    this.socket?.emit('create_duel', data)
  }

  joinDuel(code: string, nombre: string): void {
    this.socket?.emit('join_duel', { code, nombre })
  }

  ready(): void {
    this.socket?.emit('player_ready')
  }

  catchWord(wordId: string): void {
    this.socket?.emit('word_caught', wordId)
  }

  usePower(powerId: PowerId): void {
    this.socket?.emit('use_power', powerId)
  }

  requestRematch(): void {
    this.socket?.emit('request_rematch')
  }

  acceptRematch(): void {
    this.socket?.emit('accept_rematch')
  }

  get connected(): boolean {
    return this.socket?.connected ?? false
  }
}

export const duelService = new DuelService()
```

```typescript
function buildWordList(cat: CategoryKey, nivel: number): Array<{ id: string; text: string; isCorrect: boolean }> {
  const catData = CATS[cat]
  const pcCorr = nivel === 1 ? 0.70 : nivel === 2 ? 0.50 : 0.30
  const total = nivel === 1 ? 20 : nivel === 2 ? 25 : 30

  const okPool = [...catData.ok].sort(() => Math.random() - 0.5)
  const malPool = [...catData.mal].sort(() => Math.random() - 0.5)

  const numCorr = Math.round(total * pcCorr)
  const numMal = total - numCorr

  const words: Array<{ id: string; text: string; isCorrect: boolean }> = []
  okPool.slice(0, numCorr).forEach((w, i) => words.push({ id: `w${i}`, text: w, isCorrect: true }))
  malPool.slice(0, numMal).forEach((w, i) => words.push({ id: `d${i}`, text: w, isCorrect: false }))
  words.sort(() => Math.random() - 0.5)

  return words
}
```

```typescript
export function initDuelRound(config: RoundConfig): void {
  _config = config
  _words = config.words.map(w => ({ ...w, takenBy: null }))
  _scores = { A: 0, B: 0 }

// ── Board ──────────────────────────────────────────────────────────────────

function renderBoard(): void {
  const board = document.getElementById('duelWordBoard')!
  board.innerHTML = ''

  const allItems: Array<{ text: string; wordId: string | null; isDecoy: boolean }> = [
    ..._words.filter(w => w.takenBy === null).map(w => ({ text: w.text, wordId: w.id, isDecoy: false })),
    ..._activeDecoys.map(d => ({ text: d, wordId: null, isDecoy: true })),
  ]
  allItems.sort(() => Math.random() - 0.5)

  allItems.forEach(item => {
    const chip = document.createElement('div')
    chip.className = 'duel-word-chip' + (item.isDecoy ? ' decoy' : '')
    chip.textContent = item.text
    if (item.wordId) {
      chip.addEventListener('click', () => handleWordClick(item.wordId!))
    }
    board.appendChild(chip)
  })
}

function handleWordClick(wordId: string): void {
  if (_isFrozen) return
  const word = _words.find(w => w.id === wordId)
  if (!word || word.takenBy !== null) return

  // Optimistic: visually mark chip immediately
  const chips = document.querySelectorAll<HTMLElement>('.duel-word-chip')
  chips.forEach(c => {
    if (c.textContent === word.text && !c.classList.contains('taken-by-me') && !c.classList.contains('taken-by-rival')) {
      c.classList.add('taken-by-me')
    }
  })

  duelService.catchWord(wordId)
}

function applyWordTaken(wordId: string, bySlot: PlayerSlot): void {
  const word = _words.find(w => w.id === wordId)
  if (!word) return
  word.takenBy = bySlot

  const mySlot = _config!.mySlot
  const cssClass = bySlot === mySlot ? 'taken-by-me' : 'taken-by-rival'

  // Update all chips with this word text
  document.querySelectorAll<HTMLElement>('.duel-word-chip').forEach(c => {
    if (c.textContent === word.text) {
      c.classList.remove('taken-by-me', 'taken-by-rival')
      c.classList.add(cssClass)
    }
  })
}

// ── HUD ───────────────────────────────────────────────────────────────────

function updateHUD(): void {
  const mySlot = _config!.mySlot
  document.getElementById('duelMyName')!.textContent = _config!.myNombre
  document.getElementById('duelRivalName')!.textContent = _config!.rivalNombre
  document.getElementById('duelMyScore')!.textContent = String(_scores[mySlot])
  document.getElementById('duelRivalScore')!.textContent = String(_scores[mySlot === 'A' ? 'B' : 'A'])
}

function startClientTimer(): void {
  if (_timerHandle) clearInterval(_timerHandle)
  _gameEndAt = Date.now() + _config!.duracion * 1000

  const timerEl = document.getElementById('duelTimer')!
  timerEl.classList.remove('urgent')

  _timerHandle = setInterval(() => {
    const remaining = Math.max(0, _gameEndAt - Date.now())
    const secs = Math.ceil(remaining / 1000)
    const m = Math.floor(secs / 60)
    const s = secs % 60
    timerEl.textContent = `${m}:${s.toString().padStart(2, '0')}`
    if (secs <= 10) timerEl.classList.add('urgent')
    if (remaining <= 0) {
      if (_timerHandle) clearInterval(_timerHandle)
    }
  }, 500)
}

// ── Powers ────────────────────────────────────────────────────────────────

export function renderPowers(): void {
  for (let i = 0; i < 2; i++) {
    const slot = document.getElementById(`duelPower${i}`)!
    const powerId = _inventory[i] ?? null

    if (!powerId) {
      slot.className = 'duel-power-slot empty'
      slot.innerHTML = ''
      slot.onclick = null
      return
    }

    const meta = POWER_META[powerId]
    const cooldownEnd = _powerCooldowns[powerId] ?? 0
    const onCooldown = cooldownEnd > Date.now()

    slot.className = `duel-power-slot has-power${onCooldown ? ' on-cooldown' : ''}`
    slot.innerHTML = `
      <span class="duel-power-icon">${meta.icon}</span>
      <span class="duel-power-label">${meta.label}</span>
      ${onCooldown ? `<div class="duel-power-cooldown">${Math.ceil((cooldownEnd - Date.now()) / 1000)}s</div>` : ''}
    `
    slot.onclick = onCooldown ? null : () => activatePower(powerId)
  }
}

function activatePower(powerId: PowerId): void {
  const cooldownEnd = _powerCooldowns[powerId] ?? 0
  if (cooldownEnd > Date.now()) return

  duelService.usePower(powerId)
  // Optimistic cooldown start (server will confirm)
  _powerCooldowns[powerId] = Date.now() + POWER_META[powerId].cooldownMs
  renderPowers()

  // Update cooldown display every second
  const tick = setInterval(() => {
    if ((Date.now() >= (_powerCooldowns[powerId] ?? 0))) clearInterval(tick)
    renderPowers()
  }, 1000)
}

// ── Freeze overlay ────────────────────────────────────────────────────────

function applyFreeze(durationMs: number): void {
  _isFrozen = true
  _freezeEnd = Date.now() + durationMs
  const overlay = document.getElementById('duelFreezeOverlay')!
  const timerEl = document.getElementById('duelFreezeTimer')!
  overlay.style.display = 'flex'

  if (_freezeHandle) clearInterval(_freezeHandle)
  _freezeHandle = setInterval(() => {
    const remaining = Math.max(0, _freezeEnd - Date.now())
    timerEl.textContent = String(Math.ceil(remaining / 1000))
    if (remaining <= 0) {
      clearInterval(_freezeHandle!)
      _freezeHandle = null
      _isFrozen = false
      overlay.style.display = 'none'
    }
  }, 200)
}

// ── Event registration ────────────────────────────────────────────────────

function registerGameEvents(): void {
  duelService.on('word_taken', ({ wordId, bySlot, scores }) => {
    applyWordTaken(wordId, bySlot)
    updateScores(scores)
  })

  duelService.on('word_already_taken', (wordId) => {
    // Remove optimistic taken-by-me class
    const word = _words.find(w => w.id === wordId)
    if (!word) return
    document.querySelectorAll<HTMLElement>('.duel-word-chip').forEach(c => {
      if (c.textContent === word.text) c.classList.remove('taken-by-me')
    })
  })

  duelService.on('power_earned', (powerId) => {
    if (_inventory.length < 2) _inventory.push(powerId)
    renderPowers()
  })

  duelService.on('power_effect', ({ powerId, decoyWords }) => {
    if (powerId === 'FREEZE') {
      applyFreeze(3000)
    } else if (powerId === 'DECOY') {
      _activeDecoys = decoyWords
      renderBoard()
      const notice = document.getElementById('duelDecoyNotice')!
      notice.style.display = 'block'
      setTimeout(() => {
        _activeDecoys = []
        notice.style.display = 'none'
        renderBoard()
      }, 8000)
    } else if (powerId === 'STEAL') {
      const myEffects = document.getElementById('duelMyEffects')!
      const icon = document.createElement('span')
      icon.title = 'Robo activo'
      icon.textContent = '🎯'
      myEffects.appendChild(icon)
      setTimeout(() => icon.remove(), 30000)
    }
  })

  duelService.on('power_blocked', () => {
    // Visual: brief shield flash
    const flash = document.createElement('div')
    flash.textContent = '🛡️ Bloqueado'
    flash.style.cssText = 'position:fixed;top:50%;left:50%;transform:translate(-50%,-50%);background:#8b5cf6;color:#fff;padding:.5rem 1.2rem;border-radius:2rem;font-weight:700;z-index:200;pointer-events:none'
    document.body.appendChild(flash)
    setTimeout(() => flash.remove(), 1500)
  })

  duelService.on('duel_end', ({ winner, scores, reason }) => {
    if (_timerHandle) clearInterval(_timerHandle)
    if (_freezeHandle) clearInterval(_freezeHandle)
    showDuelResult({ winner, scores, mySlot: _config!.mySlot, myNombre: _config!.myNombre, rivalNombre: _config!.rivalNombre, reason })
    mostrarDuelResult()

    if (_timerHandle) clearInterval(_timerHandle)
    showDuelResult({
      winner: _config!.mySlot,
      scores: _scores,
      mySlot: _config!.mySlot,
      myNombre: _config!.myNombre,
      rivalNombre: _config!.rivalNombre,
      reason: 'disconnect',
    })
    mostrarDuelResult()
  })
}
```
